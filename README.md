## Install

```bash
npm install --save blushui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'blushui'
import 'blushui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
